01 / Overview

THE CONCEPT

The Home Security Alarm System is designed to detect unauthorized movement in a room or protected area using passive infrared (PIR) technology. When an intruder enters the detection zone, the system instantly triggers both an audible alarm and a visual LED alert.

Built on the Arduino Uno microcontroller, this project demonstrates real-world security automation ‚Äî from basic motion sensing all the way to GSM-based SMS alerts and keypad-controlled arming/disarming.

03 / How It Works

WORKING PRINCIPLE

1
PIR Sensor Monitors Continuously
The HC-SR501 PIR sensor constantly scans the protected area for changes in infrared radiation ‚Äî the heat signature produced by human bodies. It operates in idle state, consuming minimal power.
2
Motion Detected ‚Äî HIGH Signal Sent
When the sensor detects movement within its range, it sends a HIGH (5V) digital signal to Pin 2 of the Arduino. This transition from LOW to HIGH triggers the alarm sequence.
3
Arduino Triggers Alarm & LED
The Arduino reads the PIR signal, then simultaneously sets Pin 8 HIGH (buzzer ‚Äî alarm sounds) and Pin 13 HIGH (LED ‚Äî flashes red). The LCD updates to show "INTRUDER DETECTED!"
4
Optional: SMS Alert via GSM
If the SIM800L module is connected, the Arduino sends an AT command to dispatch an instant SMS to the registered number ‚Äî "‚ö† INTRUDER DETECTED at HOME!"
5
System Returns to Idle
When the PIR signal goes LOW (no motion detected), the Arduino deactivates the buzzer and LED and updates the display back to "System Armed ‚Äî Monitoring". Ready for next detection.

// ============================================
// Home Security Alarm System ‚Äî Arduino Uno
// Components: PIR Sensor, Buzzer, LED, LCD
// ============================================

#include <LiquidCrystal.h>

// Pin Definitions
const int PIR_PIN    = 2;   // PIR sensor output
const int BUZZER_PIN = 8;   // Active buzzer
const int LED_PIN    = 13;  // Alarm LED

// LCD: RS, E, D4, D5, D6, D7
LiquidCrystal lcd(7, 6, 5, 4, 3, 9);

bool alarmActive = false;

void setup() {
  pinMode(PIR_PIN, INPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);

  Serial.begin(9600);
  lcd.begin(16, 2);

  // Startup display
  lcd.setCursor(0, 0);
  lcd.print("SYSTEM ARMED");
  lcd.setCursor(0, 1);
  lcd.print("MONITORING...");

  Serial.println("[INFO] Security system initialized.");
}

void loop() {
  int motionState = digitalRead(PIR_PIN);

  if (motionState == HIGH) {
    triggerAlarm();
  } else {
    resetAlarm();
  }

  delay(200);
}

void triggerAlarm() {
  if (!alarmActive) {
    alarmActive = true;
    Serial.println("[ALERT] INTRUDER DETECTED!");

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("!! INTRUDER !!");
    lcd.setCursor(0, 1);
    lcd.print("ALARM ACTIVE");
  }

  digitalWrite(BUZZER_PIN, HIGH);
  digitalWrite(LED_PIN, HIGH);
  delay(300);
  digitalWrite(LED_PIN, LOW);  // Flash LED
  delay(200);
}

void resetAlarm() {
  if (alarmActive) {
    alarmActive = false;
    Serial.println("[INFO] All clear. System reset.");

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("SYSTEM ARMED");
    lcd.setCursor(0, 1);
    lcd.print("MONITORING...");
  }

  digitalWrite(BUZZER_PIN, LOW);
  digitalWrite(LED_PIN, LOW);
}
